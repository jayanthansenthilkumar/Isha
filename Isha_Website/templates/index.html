api.py 
from ishaa import Ishaa, Blueprint

app = Ishaa("api")
api = Blueprint("api", prefix="/api/v1")

@api.route("/users")
async def list_users(request):
    users = await User.all()
    return [{"name": u.name} for u in users]

@api.route("/users", methods=["POST"])
async def create_user(request):
    data = await request.json()
    user = await User.create(**data)
    return {"id": user.id}, 201

app.register_blueprint(api)
models.py 
from ishaa.orm import (
    Database, Model,
    TextField, IntegerField, BooleanField
)

db = Database("myapp.db")

class User(Model):
    _table = "users"
    _db = db
    name  = TextField()
    email = TextField()
    age   = IntegerField()
    admin = BooleanField(default=False)

# Create, query, update, delete
user = await User.create(name="Ishaa", email="hi@ishaa.dev")
all_users = await User.all()
secure_app.py 
from ishaa import Ishaa
from ishaa.middleware import (
    CORSMiddleware,
    SecurityHeadersMiddleware,
)

app = Ishaa("secure_app")

# Stack middleware in one line
app.add_middleware(CORSMiddleware(
    allow_origins=["https://mysite.com"]
))
app.add_middleware(SecurityHeadersMiddleware())

@app.route("/secret")
async def secret(request):
    return {"data": "Protected!"}
